python examples/fsss/build_fsss_splits.py --raw_dir data/raw/easy_clean --output_dir data/processed/easy_clean

python examples/fsss/train_ts2vec_style.py --train_file data/processed/easy_noise/train_windows.npz --val_file data/processed/easy_noise/val_windows.npz --output_dir outputs/easy_noise --epochs 80

python examples/fsss/extract_ts2vec_style_embeddings.py --train_file data/processed/easy_clean/train_windows.npz --val_file data/processed/easy_clean/val_windows.npz --test_file data/processed/easy_clean/test_windows.npz --checkpoint outputs/easy_clean/best_model.pt --output_dir outputs/easy_clean/embeddings

python examples/fsss/extract_ts2vec_style_embeddings.py --train_file data/processed/easy_noise/train_windows.npz --val_file data/processed/easy_noise/val_windows.npz --test_file data/processed/easy_noise/test_windows.npz --checkpoint outputs/easy_noise/best_model.pt --output_dir outputs/easy_noise/embeddings

python examples/fsss/extract_ts2vec_style_embeddings.py --train_file data/processed/easy_clean/train_windows.npz --val_file data/processed/easy_clean/val_windows.npz --test_file data/processed/easy_clean/test_windows.npz --checkpoint outputs/easy_clean_improve_var/best_model.pt --output_dir outputs/easy_clean_improve_var/embeddings
